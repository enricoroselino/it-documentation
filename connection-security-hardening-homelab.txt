sudo tee /etc/sysctl.d/99-game-server-homelab.conf > /dev/null <<'EOF'
# ========================================
# Game Server HOMELAB (Backend)
# 64GB RAM - Receives ALL traffic from 1 Tailscale IP
# SPECIAL: All players appear as same source IP
# For MMORPG with 24/7 AFK Grinding - Multiple Accounts
# Ubuntu 24.04
# ========================================

# === UDP Buffer Tuning (MASSIVE - many connections from 1 IP) ===
# Optimized for 64GB RAM with multiple game accounts
net.core.rmem_max = 536870912          # 512MB (doubled from 256MB)
net.core.wmem_max = 536870912          # 512MB (doubled from 256MB)
net.core.rmem_default = 67108864       # 64MB (doubled from 32MB)
net.core.wmem_default = 67108864       # 64MB (doubled from 32MB)
net.ipv4.udp_rmem_min = 16384
net.ipv4.udp_wmem_min = 16384

# UDP memory limits (pages) - massive allocation for game traffic
# 64GB RAM allows generous UDP memory (4KB per page)
net.ipv4.udp_mem = 262144 524288 1048576  # Min 1GB, Pressure 2GB, Max 4GB

# Increase max socket option memory
net.core.optmem_max = 131072

# === Network Queue & Device Tuning ===
# Higher because ALL traffic comes in bursts from 1 IP
net.core.netdev_max_backlog = 100000   # Increased from 65536 for high packet rate
net.core.netdev_budget = 3000          # Increased from 2000
net.core.netdev_budget_usecs = 20000

# Optimize for multi-core packet processing
net.core.dev_weight = 64

# Ring buffer parameters (hardware queue)
# Note: Actual ring buffer set via ethtool, this is soft limit

# === Connection Tracking (CRITICAL for port-based differentiation) ===
# VERY HIGH - kernel must track by port pairs, not just IPs
# All players from same source IP = need port-based tracking
net.netfilter.nf_conntrack_max = 3145728  # 3M (increased from 2M)

# Conntrack hash table size (CRITICAL for 1-IP scenario)
# Larger hash table = faster lookups when many connections share same source IP
# Note: Must be set via module parameter at boot
# echo "options nf_conntrack hashsize=786432" | sudo tee /etc/modprobe.d/nf_conntrack.conf
net.netfilter.nf_conntrack_buckets = 786432  # Increased from 524288

# Aggressive cleanup for non-game protocols
net.netfilter.nf_conntrack_generic_timeout = 300

# TCP timeout - extended for game launchers and chat
net.netfilter.nf_conntrack_tcp_timeout_established = 10800  # 3 hours (up from 2h)
net.netfilter.nf_conntrack_tcp_timeout_time_wait = 30
net.netfilter.nf_conntrack_tcp_timeout_close_wait = 60
net.netfilter.nf_conntrack_tcp_timeout_fin_wait = 60

# UDP: CRITICAL for 24/7 AFK grinding from same source IP
# Kernel differentiates ONLY by port pairs - long timeout essential
net.netfilter.nf_conntrack_udp_timeout = 3600             # 1 hour (up from 30min)
net.netfilter.nf_conntrack_udp_timeout_stream = 43200     # 12 hours (up from 8h)

# ICMP timeout
net.netfilter.nf_conntrack_icmp_timeout = 30

# Disable connection tracking helpers (save memory)
# Set via: echo 0 > /proc/sys/net/netfilter/nf_conntrack_helper

# === Port Range (CRITICAL - need many ports for same-IP connections) ===
# Game server needs to handle many concurrent connections from 1 source IP
# Each account connection differentiated ONLY by source/destination port pairs
net.ipv4.ip_local_port_range = 1024 65535  # Maximum available range

# === TCP Tuning (Optimized for 64GB RAM) ===
net.ipv4.tcp_syncookies = 1
net.ipv4.tcp_max_syn_backlog = 65536   # Doubled from 32k for burst handling
net.core.somaxconn = 32768             # Doubled from 16k
net.ipv4.tcp_synack_retries = 2
net.ipv4.tcp_syn_retries = 3

# TCP buffer sizes - MASSIVE for 64GB RAM
net.ipv4.tcp_rmem = 8192 262144 536870912    # Min 8KB, Default 256KB, Max 512MB
net.ipv4.tcp_wmem = 8192 262144 536870912    # Min 8KB, Default 256KB, Max 512MB

# TCP memory limits (pages) - generous for 64GB system
# Total TCP memory can use up to ~16GB (4M pages * 4KB)
net.ipv4.tcp_mem = 1048576 2097152 4194304   # Min 4GB, Pressure 8GB, Max 16GB

# TCP orphan and TIME_WAIT limits (handle many connections from same IP)
net.ipv4.tcp_max_orphans = 262144    # Increased from default
net.ipv4.tcp_max_tw_buckets = 2000000  # 2M TIME_WAIT sockets

# TCP connection lifecycle - optimized for long game sessions
net.ipv4.tcp_fin_timeout = 30
net.ipv4.tcp_keepalive_time = 600      # 10 minutes (up from 5min)
net.ipv4.tcp_keepalive_probes = 5
net.ipv4.tcp_keepalive_intvl = 30      # 30 seconds between probes

# BBR congestion control (best for high-bandwidth scenarios)
net.ipv4.tcp_congestion_control = bbr
net.core.default_qdisc = fq
net.ipv4.tcp_rfc1337 = 1

# Fast socket reuse (CRITICAL for many connections from same IP)
net.ipv4.tcp_tw_reuse = 1

# Disable slow start after idle (better for persistent game connections)
net.ipv4.tcp_slow_start_after_idle = 0

# TCP performance optimizations
net.ipv4.tcp_timestamps = 1
net.ipv4.tcp_window_scaling = 1
net.ipv4.tcp_sack = 1
net.ipv4.tcp_mtu_probing = 1

# Limit per-connection output (prevent single connection hogging)
net.ipv4.tcp_limit_output_bytes = 524288   # 512KB (up from 256KB)

# === ICMP Tuning ===
net.ipv4.icmp_echo_ignore_broadcasts = 1
net.ipv4.icmp_ratelimit = 50
net.ipv4.icmp_ignore_bogus_error_responses = 1
net.ipv4.icmp_errors_use_inbound_ifaddr = 0

# === Security (Relaxed - behind VPS + Tailscale) ===
# IMPORTANT: Relax rp_filter for asymmetric routing via Tailscale
net.ipv4.conf.all.rp_filter = 2        # Loose mode for VPN
net.ipv4.conf.default.rp_filter = 2    # Loose mode for VPN
net.ipv4.conf.tailscale0.rp_filter = 2 # Loose mode on Tailscale interface

# Ignore redirects (still important even behind VPN)
net.ipv4.conf.all.accept_redirects = 0
net.ipv4.conf.default.accept_redirects = 0
net.ipv4.conf.all.secure_redirects = 0
net.ipv4.conf.default.secure_redirects = 0

# Source routing protection
net.ipv4.conf.all.accept_source_route = 0
net.ipv4.conf.default.accept_source_route = 0

# No martian logging (1-IP VPN traffic will trigger false positives)
net.ipv4.conf.all.log_martians = 0
net.ipv4.conf.default.log_martians = 0

# Disable send redirects
net.ipv4.conf.all.send_redirects = 0
net.ipv4.conf.default.send_redirects = 0

# === IPv6 ===
net.ipv6.conf.all.disable_ipv6 = 1
net.ipv6.conf.default.disable_ipv6 = 1
net.ipv6.conf.lo.disable_ipv6 = 1

# === Performance Tuning ===
net.ipv4.tcp_retries2 = 8
net.ipv4.tcp_retries1 = 3

# === File Descriptor Limits (CRITICAL for many connections) ===
# Massive increase for handling thousands of connections from same IP
fs.file-max = 8388608  # 8M (doubled from 4M)
fs.nr_open = 8388608   # Match file-max

# Inotify limits (if running monitoring tools)
fs.inotify.max_user_watches = 524288
fs.inotify.max_user_instances = 512

# === Shared Memory (for game server applications) ===
# Generous limits for 64GB system
kernel.shmmax = 68719476736   # 64GB
kernel.shmall = 16777216      # 64GB in pages (4KB pages)

# === Process Limits ===
# High limit for multiple game instances and accounts
kernel.pid_max = 4194304      # 4M processes

# === ARP Cache (important for Tailscale mesh) ===
net.ipv4.neigh.default.gc_thresh1 = 8192   # Doubled from 4096
net.ipv4.neigh.default.gc_thresh2 = 16384  # Doubled from 8192
net.ipv4.neigh.default.gc_thresh3 = 32768  # Doubled from 16384
net.ipv4.neigh.default.gc_stale_time = 120 # 2 minutes

# === Receive Packet Steering (multi-core optimization) ===
# Distribute packet processing across CPU cores
net.core.rps_sock_flow_entries = 65536    # Doubled from 32768

# Flow table size for RFS (Receive Flow Steering)
# Helps distribute same-IP connections across cores
net.core.netdev_max_backlog = 100000

# === VM/Memory Management (64GB RAM optimization) ===
# More conservative memory management for server workload
vm.swappiness = 1                     # Almost never swap (was 10)
vm.vfs_cache_pressure = 50            # Balanced cache pressure
vm.dirty_ratio = 15                   # 15% of RAM before blocking
vm.dirty_background_ratio = 5         # Start background writeback at 5%
vm.dirty_expire_centisecs = 3000      # Dirty data lifetime: 30 seconds
vm.dirty_writeback_centisecs = 500    # Writeback every 5 seconds

# Memory overcommit (allow more allocation)
vm.overcommit_memory = 1              # Always overcommit
vm.overcommit_ratio = 50

# Minimum free memory (keep 256MB free)
vm.min_free_kbytes = 262144

# === Kernel Panic Settings ===
kernel.panic = 10                     # Reboot 10 seconds after panic
kernel.panic_on_oops = 1              # Panic on oops

# === Entropy Pool (for game RNG and crypto operations) ===
# Ensure sufficient entropy for multiple game instances
kernel.random.read_wakeup_threshold = 64
kernel.random.write_wakeup_threshold = 128
EOF

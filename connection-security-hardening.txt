sudo tee /etc/sysctl.d/99-game-server.conf > /dev/null <<'EOF'
# ========================================
# Game Server VPS (Frontend/Forwarder)
# 1 CPU / 1GB RAM - Forward traffic to homelab via Tailscale
# Ubuntu 24.04
# Public-facing forwarder optimized for single-core low memory
# ========================================

# === IPv4 Forwarding (REQUIRED for VPS forwarder) ===
net.ipv4.ip_forward = 1

# === UDP Buffer Tuning (Conservative for 1GB RAM) ===
# Smaller buffers appropriate for single-core forwarding
net.core.rmem_max = 8388608
net.core.wmem_max = 8388608
net.core.rmem_default = 1048576
net.core.wmem_default = 1048576
net.ipv4.udp_rmem_min = 4096
net.ipv4.udp_wmem_min = 4096

# === UDP Optimization for Game Traffic ===
# Prevent UDP packet drops under load
net.ipv4.udp_early_demux = 1
net.ipv4.udp_l3mdev_accept = 0

# === Network Queue & Device Tuning ===
# Reduced for single-core CPU
net.core.netdev_max_backlog = 4096
net.core.netdev_budget = 150
net.core.netdev_budget_usecs = 2000

# Optimize packet processing for single CPU
net.core.dev_weight = 32

# === Connection Tracking (LIMITED by 1GB RAM + 1 CPU) ===
# Lower capacity for single-core VPS that only forwards
net.netfilter.nf_conntrack_max = 65536

# Note: nf_conntrack_buckets must be set via module parameter:
# echo "options nf_conntrack hashsize=16384" | sudo tee /etc/modprobe.d/nf_conntrack.conf

# Aggressive timeout for unknown protocols to free memory
net.netfilter.nf_conntrack_generic_timeout = 180

# TCP timeout
net.netfilter.nf_conntrack_tcp_timeout_established = 3600  # 1 hour (was 2h)
net.netfilter.nf_conntrack_tcp_timeout_time_wait = 30
net.netfilter.nf_conntrack_tcp_timeout_close_wait = 30
net.netfilter.nf_conntrack_tcp_timeout_fin_wait = 30

# UDP: Still need reasonable timeout for forwarding
net.netfilter.nf_conntrack_udp_timeout = 300
net.netfilter.nf_conntrack_udp_timeout_stream = 3600

# ICMP timeout (lower for forwarding)
net.netfilter.nf_conntrack_icmp_timeout = 30

# Connection tracking helpers (disable unused protocols to save memory)
# These should be set via /etc/modprobe.d/ if needed

# === TCP Tuning (Conservative) ===
net.ipv4.tcp_syncookies = 1
net.ipv4.tcp_max_syn_backlog = 2048
net.core.somaxconn = 1024
net.ipv4.tcp_synack_retries = 2
net.ipv4.tcp_syn_retries = 3

# TCP buffer sizes - reduced for 1 CPU / 1GB RAM
net.ipv4.tcp_rmem = 4096 87380 8388608
net.ipv4.tcp_wmem = 4096 65536 8388608

# TCP memory limits (critical for 1GB system)
# min: below this, TCP doesn't throttle
# pressure: start applying pressure
# max: hard limit (reduced for single CPU)
net.ipv4.tcp_mem = 32768 65536 131072

# Limit orphaned connections (memory leak prevention)
net.ipv4.tcp_max_orphans = 4096
net.ipv4.tcp_max_tw_buckets = 8192

# TCP connection lifecycle
net.ipv4.tcp_fin_timeout = 20
net.ipv4.tcp_keepalive_time = 300
net.ipv4.tcp_keepalive_probes = 3
net.ipv4.tcp_keepalive_intvl = 20

# Fast recycling of TIME_WAIT sockets
net.ipv4.tcp_tw_reuse = 1

# BBR congestion control (better for high-latency forwarding)
net.ipv4.tcp_congestion_control = bbr
net.core.default_qdisc = fq

# TCP timestamps and SACK (improve forwarding performance)
net.ipv4.tcp_timestamps = 1
net.ipv4.tcp_sack = 1

# Protect against TIME_WAIT assassination
net.ipv4.tcp_rfc1337 = 1

# Disable slow start after idle (better for forwarding)
net.ipv4.tcp_slow_start_after_idle = 0

# MTU probing (useful for Tailscale tunnels)
net.ipv4.tcp_mtu_probing = 1

# === ICMP Tuning ===
net.ipv4.icmp_echo_ignore_broadcasts = 1
net.ipv4.icmp_ratelimit = 100
net.ipv4.icmp_ignore_bogus_error_responses = 1
net.ipv4.icmp_errors_use_inbound_ifaddr = 0

# === Security Hardening (PUBLIC FACING) ===
# Anti-spoofing protection (CRITICAL for public forwarder)
net.ipv4.conf.all.rp_filter = 1
net.ipv4.conf.default.rp_filter = 1

# Ignore redirects (prevent MITM)
net.ipv4.conf.all.accept_redirects = 0
net.ipv4.conf.default.accept_redirects = 0
net.ipv4.conf.all.secure_redirects = 0
net.ipv4.conf.default.secure_redirects = 0
net.ipv6.conf.all.accept_redirects = 0
net.ipv6.conf.default.accept_redirects = 0

# Don't send redirects
net.ipv4.conf.all.send_redirects = 0
net.ipv4.conf.default.send_redirects = 0

# Source routing protection (prevent routing attacks)
net.ipv4.conf.all.accept_source_route = 0
net.ipv4.conf.default.accept_source_route = 0
net.ipv6.conf.all.accept_source_route = 0
net.ipv6.conf.default.accept_source_route = 0

# Log suspicious packets (martians)
net.ipv4.conf.all.log_martians = 1
net.ipv4.conf.default.log_martians = 1

# Ignore ICMP echo requests (optional - disable ping to public IP)
# net.ipv4.icmp_echo_ignore_all = 1

# SYN flood protection
net.ipv4.tcp_syncookies = 1
net.ipv4.tcp_max_syn_backlog = 4096

# Reverse path filtering for Tailscale interface (may need adjustment)
# If Tailscale has issues, set tailscale0 to loose mode:
# net.ipv4.conf.tailscale0.rp_filter = 2

# === IPv6 ===
net.ipv6.conf.all.disable_ipv6 = 1
net.ipv6.conf.default.disable_ipv6 = 1
net.ipv6.conf.lo.disable_ipv6 = 1

# === Performance Tuning (Memory-conscious) ===
net.ipv4.ip_local_port_range = 15000 65535
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_retries2 = 6

# Reduce TCP retransmission overhead
net.ipv4.tcp_retries1 = 3

# ARP cache tuning (reduce memory for ARP table)
net.ipv4.neigh.default.gc_thresh1 = 128
net.ipv4.neigh.default.gc_thresh2 = 512
net.ipv4.neigh.default.gc_thresh3 = 1024
net.ipv4.neigh.default.gc_stale_time = 60

# Route cache (if applicable)
net.ipv4.route.gc_timeout = 100

# === Memory Pressure Management ===
# More aggressive memory reclaim for 1GB system
vm.min_free_kbytes = 65536
vm.swappiness = 10
vm.vfs_cache_pressure = 50

# Prevent OOM killer from being too aggressive
vm.overcommit_memory = 1
vm.overcommit_ratio = 50

# Dirty page management (write to disk sooner to free RAM)
vm.dirty_ratio = 10
vm.dirty_background_ratio = 5
vm.dirty_expire_centisecs = 1000
vm.dirty_writeback_centisecs = 500

# === Additional Forwarding Optimizations ===
# Disable proxy ARP (security - don't respond to ARP for other hosts)
net.ipv4.conf.all.proxy_arp = 0
net.ipv4.conf.default.proxy_arp = 0

# IP forwarding with RP filter consideration
# If using asymmetric routing (common with Tailscale), may need:
# net.ipv4.conf.all.rp_filter = 2
# net.ipv4.conf.default.rp_filter = 2

# === Kernel Panic Settings (Auto-reboot on critical errors) ===
kernel.panic = 10
kernel.panic_on_oops = 1

# === Process Limits ===
# Reduce max PIDs to save memory (single CPU)
kernel.pid_max = 8192

# File descriptor limits (reduced for single-core forwarding)
fs.file-max = 32768

# Inotify limits (reduce if not using file watching)
fs.inotify.max_user_watches = 4096
fs.inotify.max_user_instances = 64
EOF
